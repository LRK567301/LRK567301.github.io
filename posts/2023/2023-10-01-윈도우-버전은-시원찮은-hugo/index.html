version: '3.3'
services:
    hugo:
        volumes:
            - 'E:\docker\hugo\src:/src'
            - 'E:\Documents\blog\posts\public:/src/blog/content/post'
        image: 'klakegg/hugo:ext-alpine'
        command: '--source /src/blog --destination /src/blog/public'

결국 WSL 도커용 일회성 빌드 compose만 새로 짰어. 글 썼을 때만 빌드하면 되니까 계속 실행해두는 것 자체가 낭비긴 했지만 뭔가뭔가해. WSL 참….윈도우 참…..WSA 포함해서 진짜 안 되는 게 없는 미친 확장성이긴 한데 네이티브는 과연 네이티브라 벽이랄까 선이 확실해.

휴고 버전 문제였다. 윈도우 빌드도 도커랑 같이 v0.111.3으로 맞춰주니까 되네. 테마나 설정 버전이 이제 안 맞나봐. 휴고 문서에 도커용으로 걸려 있는 klakegg/hugo가 벌써 5개월 정도 버려졌다.
